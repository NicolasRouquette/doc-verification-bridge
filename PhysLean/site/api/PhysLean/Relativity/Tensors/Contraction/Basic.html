    noncomputable def TensorSpecies.Tensor.contrT {k : Type} [CommRing k] {C G : Type} [Group G] {S : TensorSpecies k C G} (n : ℕ) {c : Fin (n + 1 + 1) → C} (i j : Fin (n + 1 + 1)) (hij : i ≠ j ∧ S.τ (c i) = c j) :
    S.Tensor c →ₗ[k] S.Tensor (c ∘ Pure.dropPairEmb i j)

    For c : Fin (n + 1 + 1) → C, i j : Fin (n + 1 + 1) with dual color, and a tensor t : Tensor S c, contrT i j _ t is the tensor formed by contracting the ith index of t with the jth index.
