    Boosts applied to Lorentz vectors #

    These recover what one would describe as the ordinary Lorentz transformations of Lorentz vectors.

    source
    theorem Lorentz.Vector.boost_time_eq {d : ℕ} (i : Fin d) (β : ℝ) (hβ : |β| < 1) (p : Vector d) :
    (LorentzGroup.boost i β hβ • p) (Sum.inl 0) = LorentzGroup.γ β * (p (Sum.inl 0) - β * p (Sum.inr i))
    source
    theorem Lorentz.Vector.boost_inr_self_eq {d : ℕ} (i : Fin d) (β : ℝ) (hβ : |β| < 1) (p : Vector d) :
    (LorentzGroup.boost i β hβ • p) (Sum.inr i) = LorentzGroup.γ β * (p (Sum.inr i) - β * p (Sum.inl 0))
    source
    theorem Lorentz.Vector.boost_inr_other_eq {d : ℕ} (i j : Fin d) (hji : j ≠ i) (β : ℝ) (hβ : |β| < 1) (p : Vector d) :
    (LorentzGroup.boost i β hβ • p) (Sum.inr j) = p (Sum.inr j)
    source
    theorem Lorentz.Vector.boost_toCoord_eq {d : ℕ} (i : Fin d) (β : ℝ) (hβ : |β| < 1) (p : Vector d) :
    LorentzGroup.boost i β hβ • p = fun (j : Fin 1 ⊕ Fin d) => match j with | Sum.inl 0 => LorentzGroup.γ β * (p (Sum.inl 0) - β * p (Sum.inr i)) | Sum.inr j => if j = i then LorentzGroup.γ β * (p (Sum.inr i) - β * p (Sum.inl 0)) else p (Sum.inr j)
